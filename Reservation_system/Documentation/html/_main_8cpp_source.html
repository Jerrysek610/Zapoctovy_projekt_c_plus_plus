641  if (input[i] == '[' && !in) {
642  ++bracket;
643  in = true;
644  }
645  //zpracovani prvniho ciselneho a druheho ciselneho argumentu (zpracovani souradnic)
646  else if (input[i] == ';'&&in && (first&&!second || !first && !second)) {
647  if (!first && !second)
648  x = std::stoull(tmp);
649  else
650  y= std::stoull(tmp);
651  auto temp = first;
652  first = !first && !second ? true : false;
653  second = temp && !second ? true : false;

655  }

658  coordinates a;

664  seats.clear();
665  cost.clear();
666  return false;
667  }
668  seats.push_back(action->Get_Array_Position(a));
669  cost.push_back(std::stoi(tmp));
670  --bracket;
671  in=false;
672  second=false;
673  tmp="";
674  }
675  else if(in&&isdigit(input[i])){
676  tmp+=input[i];
677  }

679  std::cout << "spatny tvar vstupu pro nacteni sedadel." << std::endl;

682  return false;
683  }
